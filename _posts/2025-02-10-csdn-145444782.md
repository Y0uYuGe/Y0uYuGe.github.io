# IMU 惯性测量单元消息包
[https://docs.ros.org/en/noetic/api/sensor_msgs/html/msg/](https://docs.ros.org/en/noetic/api/sensor_msgs/html/msg/)
[https://docs.ros.org/en/noetic/api/sensor_msgs/html/msg/Imu.html](https://docs.ros.org/en/noetic/api/sensor_msgs/html/msg/Imu.html)

sensor_msgs/Imu 消息用于表示 IMU 数据，包括方向、角速度和线性加速度等信息。通过订阅该消息，可以获取 IMU 传感器的数据，并用于机器人导航、姿态估计等应用。

1.  Header
Header：包含时间戳和坐标系信息。
stamp：时间戳，表示数据的采集时间。
frame_id：坐标系的名称，表示数据的参考坐标系。
2.  orientation
geometry_msgs/Quaternion：表示物体的方向，使用四元数表示。 [四元数如何控制物体旋转？](https://www.bilibili.com/video/BV14t421h7M4/?spm_id_from=333.337.search-card.all.click&vd_source=a1be939c65919194c77b8a6a36c14a6e)
x：四元数的 x 分量。
y：四元数的 y 分量。
z：四元数的 z 分量。
w：四元数的 w 分量。
3.  orientation_covariance
float64[9]：表示方向测量的协方差矩阵，9 个元素表示 3x3 的协方差矩阵，按行排列。
orientation_covariance[0]：xx 方向的协方差。
orientation_covariance[1]：xy 方向的协方差。
orientation_covariance[2]：xz 方向的协方差。
orientation_covariance[3]：yx 方向的协方差。
orientation_covariance[4]：yy 方向的协方差。
orientation_covariance[5]：yz 方向的协方差。
orientation_covariance[6]：zx 方向的协方差。
orientation_covariance[7]：zy 方向的协方差。
orientation_covariance[8]：zz 方向的协方差。
4. angular_velocity
geometry_msgs/Vector3：表示物体的角速度，单位为 rad/s。
x：x 轴方向的角速度。
y：y 轴方向的角速度。
z：z 轴方向的角速度。
5.  angular_velocity_covariance
float64[9]：表示角速度测量的协方差矩阵，9 个元素表示 3x3 的协方差矩阵，按行排列。
angular_velocity_covariance[0]：xx 方向的协方差。
angular_velocity_covariance[1]：xy 方向的协方差。
angular_velocity_covariance[2]：xz 方向的协方差。
angular_velocity_covariance[3]：yx 方向的协方差。
angular_velocity_covariance[4]：yy 方向的协方差。
angular_velocity_covariance[5]：yz 方向的协方差。
angular_velocity_covariance[6]：zx 方向的协方差。
angular_velocity_covariance[7]：zy 方向的协方差。
angular_velocity_covariance[8]：zz 方向的协方差。
6. linear_acceleration
geometry_msgs/Vector3：表示物体的线性加速度，单位为 m/s²。
x：x 轴方向的加速度。
y：y 轴方向的加速度。
z：z 轴方向的加速度。
7. linear_acceleration_covariance
float64[9]：表示线性加速度测量的协方差矩阵，9 个元素表示 3x3 的协方差矩阵，按行排列。
linear_acceleration_covariance[0]：xx 方向的协方差。
linear_acceleration_covariance[1]：xy 方向的协方差。
linear_acceleration_covariance[2]：xz 方向的协方差。
linear_acceleration_covariance[3]：yx 方向的协方差。
linear_acceleration_covariance[4]：yy 方向的协方差。
linear_acceleration_covariance[5]：yz 方向的协方差。
linear_acceleration_covariance[6]：zx 方向的协方差。
linear_acceleration_covariance[7]：zy 方向的协方差。
linear_acceleration_covariance[8]：zz 方向的协方差。
8. 注意事项
协方差矩阵：如果协方差矩阵中的所有元素都为零，表示协方差未知，需要从其他来源获取协方差信息。
无估计值：如果某个数据元素没有估计值（例如，IMU 不提供方向估计），应将该协方差矩阵的第一个元素设置为 -1。

```cpp
#include <ros/ros.h>
#include <sensor_msgs/Imu.h>
#include <tf/tf.h> // 四元数转换为欧拉角

/**
 * IMUCallback 函数用于处理IMU数据回调。
 * 该函数接收IMU消息，检查方向估计的有效性，并将四元数转换为欧拉角后打印出来。
 *
 * @param msg 接收到的IMU消息，包含四元数和其他传感器数据。
 */
void IMUCallback(sensor_msgs::Imu msg)
{
    // 检查是否有有效的方向估计值
    if (msg.orientation_covariance[0] < 0) 
        return;

    // 创建 tf::Quaternion 对象，用于后续将四元数转换为欧拉角
    tf::Quaternion quaternion(
        msg.orientation.x,
        msg.orientation.y,
        msg.orientation.z,
        msg.orientation.w
    );

    // 将四元数转换为欧拉角（roll, pitch, yaw）
    double roll, pitch, yaw;
    tf::Matrix3x3(quaternion).getRPY(roll, pitch, yaw);
    // tf::Matrix3x3 是一个 3x3 的矩阵类，用于表示旋转矩阵。
    // tf::Matrix3x3(quaternion)：将四元数 quaternion 转换为旋转矩阵。

    // getRPY 方法从旋转矩阵中提取欧拉角（roll、pitch、yaw）。
    // 将欧拉角从弧度转换为角度 将弧度乘以π/180，即可将弧度转换为角度。
    roll = roll * 180 / M_PI;
    pitch = pitch * 180 / M_PI;
    yaw = yaw * 180 / M_PI;

    // 打印转换后的欧拉角信息
    ROS_INFO("roll滚转: %.2f, pitch俯仰: %.2f, yaw朝向: %.2f", roll, pitch, yaw);
}

/**
 * main 函数是程序的入口点。
 * 初始化ROS节点并订阅IMU数据，进入消息循环等待回调处理。
 *
 * @param argc 命令行参数数量。
 * @param argv 命令行参数数组。
 * @return 返回0表示程序成功结束。
 */
int main(int argc, char *argv[])
{
    // 设置区域，以便正确显示字符
    setlocale(LC_ALL, "");

    // 初始化ROS节点
    ros::init(argc, argv, "imu_node");

    // 创建节点句柄
    ros::NodeHandle n;

    // 订阅IMU数据
    ros::Subscriber imu_sub = n.subscribe("/imu/data", 10, IMUCallback);

    // 进入消息循环，等待回调处理
    ros::spin();

    // 返回成功
    return 0;
}
```

```cpp
#include <ros/ros.h>
#include <sensor_msgs/Imu.h>
#include <tf/tf.h> // 四元数转换为欧拉角
#include <geometry_msgs/Twist.h>
/**
 * IMUCallback 函数用于处理IMU数据回调。
 * 该函数接收IMU消息，检查方向估计的有效性，并将四元数转换为欧拉角后打印出来。
 *
 * @param msg 接收到的IMU消息，包含四元数和其他传感器数据。
 */
ros::Publisher vel_pub;
void IMUCallback(sensor_msgs::Imu msg)
{
    // 检查是否有有效的方向估计值
    if (msg.orientation_covariance[0] < 0) 
        return;

    // 创建 tf::Quaternion 对象，用于后续将四元数转换为欧拉角
    tf::Quaternion quaternion(
        msg.orientation.x,
        msg.orientation.y,
        msg.orientation.z,
        msg.orientation.w
    );

    // 将四元数转换为欧拉角（roll, pitch, yaw）
    double roll, pitch, yaw;
    tf::Matrix3x3(quaternion).getRPY(roll, pitch, yaw);
    // tf::Matrix3x3 是一个 3x3 的矩阵类，用于表示旋转矩阵。
    // tf::Matrix3x3(quaternion)：将四元数 quaternion 转换为旋转矩阵。

    // getRPY 方法从旋转矩阵中提取欧拉角（roll、pitch、yaw）。
    // 将欧拉角从弧度转换为角度 将弧度乘以π/180，即可将弧度转换为角度。
    roll = roll * 180 / M_PI;
    pitch = pitch * 180 / M_PI;
    yaw = yaw * 180 / M_PI;

    // 打印转换后的欧拉角信息
    ROS_INFO("roll滚转: %.2f, pitch俯仰: %.2f, yaw朝向: %.2f", roll, pitch, yaw);

    double target_yaw = 90;
    double diff_angle=target_yaw-yaw;
    geometry_msgs::Twist vel_cmd;
    vel_cmd.angular.z=diff_angle*0.01;
    vel_cmd.linear.x=0.1;
    vel_pub.publish(vel_cmd);

}

int main(int argc, char *argv[])
{
    // 设置区域，以便正确显示字符
    setlocale(LC_ALL, "");

    // 初始化ROS节点
    ros::init(argc, argv, "imu_node");

    // 创建节点句柄
    ros::NodeHandle n;

    // 订阅IMU数据
    ros::Subscriber imu_sub = n.subscribe("/imu/data", 10, IMUCallback);

    vel_pub=n.advertise<geometry_msgs::Twist>("/cmd_vel", 10);

    // 进入消息循环，等待回调处理
    ros::spin();

    // 返回成功
    return 0;
}
```

```python
#! /usr/bin/env python3
#coding=utf-8
import rospy
from sensor_msgs.msg import Imu
from tf.transformations import euler_from_quaternion
import math
from geometry_msgs.msg import Twist

def imu_callback(msg):
    if msg.orientation_covariance[0] < 0:
        return
    quaternion=[
        msg.orientation.x,
        msg.orientation.y,
        msg.orientation.z,
        msg.orientation.w
    ]
    (roll,pitch,yaw)=euler_from_quaternion(quaternion)
    roll=roll*180/math.pi
    pitch=pitch*180/math.pi
    yaw=yaw*180/math.pi
    rospy.loginfo("roll滚转=%.0f pitch俯仰=%.0f yaw朝向=%.0f",roll,pitch,yaw)
    target_yaw=90
    diff_angle=target_yaw-yaw
    vel_cmd=Twist()
    vel_cmd.angular.z=diff_angle*0.01
    vel_cmd.linear.x=0.1
    global vel_pub
    vel_pub.publish(vel_cmd)
if __name__ == '__main__':
    rospy.init_node('imu_node')
    imu_sub = rospy.Subscriber('/imu/data', Imu, imu_callback,queue_size=10)
    vel_pub = rospy.Publisher('/cmd_vel', Twist, queue_size=10)
    rospy.spin()
```

# 栅栏地图格式
[https://docs.ros.org/en/api/nav_msgs/html/msg/OccupancyGrid.html](https://docs.ros.org/en/api/nav_msgs/html/msg/OccupancyGrid.html)
地图原点是左下角
![在这里插入图片描述](/images/csdn/2025/csdn-145444782/image-013-805e2e0891.png)
nav_msgs/OccupancyGrid 是 ROS（Robot Operating System）中的一种消息类型，用于表示二维网格地图（2D grid map）。每个网格单元（cell）表示该位置被占用的范围


1. Header header
类型：std_msgs/Header
作用：包含时间戳和坐标系信息，用于描述地图数据的时间和空间参考。
字段：
uint32 seq：序列号，用于标识消息的顺序。
time stamp：时间戳，表示消息的时间。
string frame_id：坐标系的名称，表示地图数据所在的参考坐标系。
2. MapMetaData info
类型：nav_msgs/MapMetaData
作用：包含地图的元数据，描述地图的属性和配置。
字段：
float32 map_load_time：地图加载的时间戳。
float32 resolution：地图的分辨率（单位：米/像素）或者说 栅格的长度
int32 width：地图的宽度（单位：像素）。
int32 height：地图的高度（单位：像素）。
geometry_msgs/Pose origin：地图的原点位置和方向，相对于坐标系的偏移。
3. int8[] data
类型：int8[]（整数数组）
作用：存储地图的实际数据，按行优先顺序排列（row-major order）。
值的含义：
-1：未知（unknown），表示该位置的占用状态未知。
0：空闲（free），表示该位置未被占用。
100：占用（occupied），表示该位置被占用。
范围：占用范围在 [0, 100] 之间，未知为 -1。

# 发布地图
![在这里插入图片描述](/images/csdn/2025/csdn-145444782/image-014-0cbc116265.png)

```cpp
#include <ros/ros.h>
#include <nav_msgs/OccupancyGrid.h>

int main(int argc, char  *argv[])
{
    ros::init(argc, argv, "map_pub_node");
    ros::NodeHandle nh;
    ros::Publisher map_pub = nh.advertise<nav_msgs::OccupancyGrid>("/map", 1);
    ros::Rate loop_rate(10);
    while(ros::ok())
    {
        nav_msgs::OccupancyGrid msg;
        msg.header.frame_id = "map";
        msg.header.stamp = ros::Time::now();
        //地图相对坐标系原点的偏移量
        msg.info.origin.position.x = 1.0; 
        msg.info.origin.position.y = 0.0;
        msg.info.resolution = 1.0; //栅格的长度
        msg.info.width = 4;
        msg.info.height = 2;

        msg.data.resize(msg.info.width * msg.info.height);
        msg.data[0]=90;
        msg.data[1]=100;
        msg.data[2]=0;
        msg.data[3]=-1;
        map_pub.publish(msg);
        loop_rate.sleep();

    }
    return 0;
}

```

```python
import rospy

if __name__ == '__main__':
    rospy.init_node('map_pub_node')
    pub = rospy.Publisher('/map', OccupancyGrid, queue_size=10)
    rate = rospy.Rate(10)
    while not rospy.is_shutdown():
        msg = OccupancyGrid()
        msg.header.stamp = rospy.Time.now()

        msg.info.resolution = 1
        msg.info.width = 4
        msg.info.height = 2
        msg.info.origin.position.x = 0

        msg.data = [0]*4*2
        msg.data[0]=100
        msg.data[1]=90
        msg.data[2]=0
        msg.data[3]=-1

        pub.publish(msg)
```
